\begin{Slide}{Contextualização}
		\begin{itemize}
		    \item Responsabilidade dos simuladores.
		    \item Ambientes 3D e o aumento da imersão nos simuladores.
		    \item Usuários remotos $\to$\ o surgimento dos AVDs.
		    \item IA contribuindo para o aumento da realidade nos simuladores.
		    \item Simulador provendo um ambiente atrativo para desenvolvimento e testes de agentes BDI.
		\end{itemize}
\end{Slide}

\begin{Slide}{Objetivos do Trabalho}
		\begin{itemize}
		    \item Desenvolver um simulador 3D de um AVD para atuar em uma rede local. 
		    \item Objetivos específicos:
		    \begin{itemize}
		    	\item Definir um cenário a ser utilizado no simulador (ambiente, objetos, percepções e comportamentos);
		    	\item Utilizar a engine gráfica JMonkey Engine para construção do ambiente gráfico do AVD;
		    	\item Permitir que os tanques sejam controlados por avatares ou tenham um comportamento autônomo.
		    \end{itemize}
		\end{itemize}
\end{Slide}

\begin{Slide}{Ambiente Virtual Distribuído}
    \begin{itemize}
    		\item Primeiros AVDs desenvolvidos:
    		\begin{itemize}
    				\item Simulator Network (SIMNET);
    				\item Naval Postgraduate School Network (NPSNET).
    		\end{itemize}
    		\item Resultados obtidos a partir desses projetos.
    		\item Modelos gerais de comunicação:
    		\begin{itemize}
    				\item Centralizado;
    				\item Distribuído.
    		\end{itemize}
    \end{itemize}
    \newpage
		\begin{itemize}
		    \item Java Game Networking (JGN)
		    \begin{itemize}
		    		\item \textsl{Open-source}.
		    		\item Maior \emph{abstração} na implementação do módulo de comunicação.
		    		\item Mensagens definidas como \emph{classes Java}.
		    		\item Faz uso dos protocolos TCP e UDP.
		    		\item Suporta UDP \textsl{broadcasting}.
		    		\item Falta de documentação $\to$\ ponto negativo.
		    \end{itemize}
		\end{itemize}
\end{Slide}

\begin{Slide}{JMonkey Engine (JME)}
		\begin{itemize}
				\item \textsl{Open-source}.
				\item Escrita totalmente em Java delegando tarefas de renderização à biblioteca nativa OpenGL.
				\item Baseada em Grafo de Cena.
				\item Sem acoplamento com uma implementação especifica de OpenGL para Java (através de JNI)
				\item Física usando o \textsl{framework} JME Physics.
				\item Conceito de \textsl{Game States}.
		\end{itemize}		
		\begin{center}
				\vspace{2pt}
				\hspace{-12pt}
				\includegraphics[width=8cm]{jme_overview.png}
		\end{center}
\end{Slide}

\begin{Slide}{Principais Requisitos}
		 Funcionais:
		 \begin{itemize}
		     \item Permitir a escolha do mapa da batalha.
		     \item Possibilitar a entrada de novos modelos de mapas.
		     \item Disponibilizar os dois modelos de tanques de guerra.
		     \item Fornecer um conjunto de ações, percepções e crenças para os agentes.
		     \item Permitir que os tanques sejam controlados por usuários ou tenham um comportamento autônomo.
		     \item Prover um AVD com arquitetura híbrida para LAN.
		 \end{itemize}
		 \newpage
		 Não-Funcionais:
		 \begin{itemize}
		 		 \item Tratar no máximo seis tanques de guerra em cada time.
				 \item Garantir que nenhum usuário se conecte ao servidor após o início da batalha.
				 \item JavaSE 6.0, JMonkey Engine, JME Physics, JGN e Jason.
		 \end{itemize}
\end{Slide}

		\textbf{Terreno}
		\begin{itemize}

\begin{Slide}{Implementação}
		\textbf{Técnicas e Ferramentas Utilizadas}
		\begin{itemize}
				\item IDE Eclipse 3.3 $\to$\ Java.
				\item Blender 2.43 $\to$\ Modelos dos Tanques 3D.
				\item Adobe Photoshop CS $\to$\ Texturas do terreno.
				\item Assembla Space $\to$\ Gerenciamento do projeto.
				\item Apache Ant $\to$\ Tarefas repetitivas.
				\item Padrões de Projeto: \textsl{Singleton}, \textsl{Command}, \textsl{Factory Method} e \textsl{Transfer Object}.
		\end{itemize}
		\newpage
		\textbf{Operacionalidade da Implementação}\linebreak
		\linebreak
		Passos:
		\begin{itemize}
				\item Abrir aplicativo servidor, configurar e iniciar a batalha.
				\item Abrir um aplicativo cliente:
				\begin{itemize}
						\item Jogador \emph{Avatar} ou \emph{MAS}.
				\end{itemize}
				\item Conectar-se ao servidor criado.
				\item Preparar os tanques.
				\item Definir-se como \emph{Ready}.
		\end{itemize}
		
		Servidor notifica inicio da batalha $\to$\ clientes carregam cenário (estado AVD local)
		
		\textbf{Jogador Avatar}
		\begin{itemize}
				\item Teclas:
				\begin{itemize}
						\item Movimentar Tanque: \textbf{W}, \textbf{S}, \textbf{A} e \textbf{D}.
						\item Atirar Arma Principal: \textbf{Barra Espaço}.
						\item Atirar Metralhadora: \textbf{Ctrl}.
						\item Movimentar Arma principal: \textbf{Direita} e \textbf{Esquerda}.
						\item Macaco hidráulico: \textbf{J}
						\item Modo Wireframed: \textbf{O}
				\end{itemize}
		\end{itemize}
		\newpage
		\textbf{Jogador MAS}
		\begin{itemize}
				\item Teclas:
				\begin{itemize}
						\item Trocar foco da camera: \textbf{TAB}
						\item Modo Wireframed: \textbf{O}
				\end{itemize}
		\end{itemize}

\begin{Slide}{Resultados e Discussão}
		\begin{itemize}
				\item \textsl{Middleware} JGN $\to$\ performance e abstração.
				\item \textsl{Engine} JME $\to$\ Grafo de cena = performance, otimização e abstração.
				\item Cenário 3D com bom nível de imersão:
				\begin{itemize}
					\item Gravidade, massa, colisão, suspensões e disparo de projéteis.
				\end{itemize}
				\item \textsl{Chase Camera} melhor visão para o jogador Avatar.
				\item Jason 1.1.1 $\to$\ desempenho nas ações.
				\item Abstração para o desenvolvedor AgentSpeak(L).
				\item \emph{Orientação a Objetos} em aplicações gráficas.
		\end{itemize}
\end{Slide}
